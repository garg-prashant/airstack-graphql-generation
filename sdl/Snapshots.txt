  _and: [SnapshotFilter!]
  _or: [SnapshotFilter!]

  tokenId: String_Comparator_Exp
  tokenAddress: Address_Comparator_Exp
  tokenType: TokenType_Comparator_Exp

  timestamp: Time_Range_Comparator_Exp
  blockNumber: Range_Comparator_Exp
}

  filter: SnapshotFilter!
